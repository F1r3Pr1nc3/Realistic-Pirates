# Add resources to each merc clan's budget yearly
event = {
	id = tob_budget.1
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		every_country = {
			limit = { is_country_type = outer_bulwarks }
			change_variable = { which = bulwarkbudget value = 200 }
		}
	}
}
