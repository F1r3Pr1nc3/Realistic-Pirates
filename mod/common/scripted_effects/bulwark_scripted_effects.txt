# this = owner of fleet
# from = fleet
# fromfrom = planet (if any)
bulwark_exploration_effect = {
	log = "triggered bulwark_exploration_effect"

	owner = {
		save_event_target_as = bulwark_eye_owner
	}
	solar_system = {
		random_neighbor_system = {
			limit = {
				NOR = {
					AND = {
						exists = owner
						owner = {
							is_fallen_empire = yes
							NOT = {
								has_authority = auth_machine_intelligence
							}
						}
					}
					any_fleet_in_system = {
						owner = {
							is_hostile_to = event_target:bulwark_eye_owner
						}
					}
				}
			}
			random_system_planet = {
				save_event_target_as = bulwark_exploration_planet_target
			}
		}
	}
	auto_move_to_planet = {
		target = event_target:bulwark_exploration_planet_target
		clear_auto_move_on_arrival = no
		arrival_effect = bulwark_exploration_effect
	}
}
